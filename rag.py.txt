import json

def load_knowledge():
    with open("data/knowledge_base.json", "r") as f:
        return json.load(f)

def retrieve_answer(query):
    kb = load_knowledge()

    if "price" in query or "plan" in query:
        return kb["pricing"]

    if "refund" in query or "support" in query:
        return kb["policies"]

    return "I can help you with pricing and plans."
